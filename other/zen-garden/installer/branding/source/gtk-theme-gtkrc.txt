# Zen Garden GTK2 Theme Configuration
# This file will be placed at /usr/share/themes/zen-garden/gtk-2.0/gtkrc in the initrd
# Format: GTK2 theme configuration (gtkrc syntax)
#
# Final Color Palette: Refined Zen Garden aesthetic
# - Neutral backgrounds with subtle green tints
# - Green accents ONLY for selection, progress, focus (premium feel)
# - High readability with proper contrast

# Pixmap path for background images
pixmap_path "/usr/share/graphics"

# --- Zen Garden Final Color Palette ---
# Background/Surfaces:
#   Primary background:            #EFF3EC (239,243,236) - subtle green tint
#   Hover state:                   #E6EDE4 (230,237,228)
#   Pressed state:                 #DDE6DD (221,230,221)
#   Input fields (elevated):       #F6FAF4 (246,250,244) - brighter than window
#
# Greens (ACCENT ONLY - selection/progress/focus):
#   Selection (primary):           #3F6B58 (63,107,88) - saturated green
#   Selection (subdued alt):       #456455 (69,100,85) - greyer, Debian-like
#   Progress/Success:              #6DA278 (109,162,120) - brand color in motion
#
# Text:
#   Primary text:                  #1F241E (31,36,30) - dark green-black
#   Inverted text (on selection):  #F7FBF6 (247,251,246) - off-white
#   Disabled text:                 #6E776C (110,119,108) - muted gray-green
#
# Designer Notes:
#   - Reserve greens for interactive states only (not large surfaces)
#   - Input fields slightly brighter than background (subtle elevation)
#   - Most UI stays neutral; greens appear only when user interacts

gtk_color_scheme = " \
  base_color:#F6FAF4 \
  bg_color:#EFF3EC \
  fg_color:#1F241E \
  text_color:#1F241E \
  selected_bg_color:#3F6B58 \
  selected_fg_color:#F7FBF6 \
"

# --- Default Style (All Widgets) ---
style "zen-garden-default"
{
  bg[NORMAL]      = "#EFF3EC"  # Primary background (subtle green tint)
  bg[PRELIGHT]    = "#E6EDE4"  # Hover state
  bg[ACTIVE]      = "#DDE6DD"  # Pressed state
  bg[SELECTED]    = "#3F6B58"  # Selection (saturated green accent)
  bg[INSENSITIVE] = "#DDE6DD"  # Disabled state

  fg[NORMAL]      = "#1F241E"  # Primary text (dark green-black)
  fg[ACTIVE]      = "#1F241E"
  fg[PRELIGHT]    = "#1F241E"
  fg[SELECTED]    = "#F7FBF6"  # Inverted text (off-white on green)
  fg[INSENSITIVE] = "#6E776C"  # Disabled text (muted gray-green)

  base[NORMAL]    = "#F6FAF4"  # Input fields (brighter = subtle elevation)
  base[SELECTED]  = "#3F6B58"  # Selected text background
  text[NORMAL]    = "#1F241E"  # Text field text
  text[SELECTED]  = "#F7FBF6"  # Selected text color
  
  # Optional: Solid background color image (if zen-bg.png exists as solid #EFF3EC)
  # bg_pixmap[NORMAL] = "zen-bg.png"
}

# --- Window Style ---
style "zen-garden-window"
{
  bg[NORMAL] = "#EFF3EC"  # Same as primary background (cohesive)
}

# --- Button Style ---
# Note: Keep buttons neutral; green appears only on hover (premium feel)
style "zen-garden-button"
{
  bg[NORMAL]   = "#E6EDE4"  # Neutral button surface
  bg[PRELIGHT] = "#3F6B58"  # Green accent on hover (brand interaction)
  bg[ACTIVE]   = "#DDE6DD"  # Pressed state
  fg[NORMAL]   = "#1F241E"  # Dark text on neutral
  fg[PRELIGHT] = "#F7FBF6"  # White text on green hover
  fg[ACTIVE]   = "#1F241E"  # Dark text when pressed
}

# --- Progress Bar Style ---
# Progress uses brand green (#6DA278) for "color in motion"
style "zen-garden-progressbar"
{
  bg[PRELIGHT] = "#6DA278"  # Brand green (progress/success color)
  fg[PRELIGHT] = "#1F241E"  # Dark text on light green
  
  # Alternative: More saturated selection green for dramatic progress
  # bg[PRELIGHT] = "#3F6B58"
  # fg[PRELIGHT] = "#F7FBF6"
}

# --- Entry/Text Field Style ---
# Inputs are brighter than background (subtle elevation = clarity)
style "zen-garden-entry"
{
  base[NORMAL]   = "#F6FAF4"  # Elevated input background
  base[SELECTED] = "#3F6B58"  # Green selection background
  text[NORMAL]   = "#1F241E"  # Input text
  text[SELECTED] = "#F7FBF6"  # Selected text
}

# --- Scrollbar Style ---
style "zen-garden-scrollbar"
{
  bg[NORMAL]   = "#E6EDE4"  # Scrollbar track
  bg[PRELIGHT] = "#3F6B58"  # Green on hover (brand interaction)
  bg[ACTIVE]   = "#DDE6DD"  # Scrollbar dragging
}

# --- Apply Styles to Widget Classes ---
class "*" style "zen-garden-default"
widget_class "*GtkWindow*" style "zen-garden-window"
widget_class "*GtkButton*" style "zen-garden-button"
widget_class "*GtkProgress*" style "zen-garden-progressbar"
widget_class "*GtkEntry*" style "zen-garden-entry"
widget_class "*GtkScrollbar*" style "zen-garden-scrollbar"
